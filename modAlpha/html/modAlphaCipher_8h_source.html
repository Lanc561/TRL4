140inline bool modAlphaCipher::isValidChar(wchar_t c) {
141 // Проверяем только русские буквы
142 return (c >= L'А' && c <= L'Я') || (c >= L'а' && c <= L'я') || c == L'Ё' || c == L'ё';

145inline bool modAlphaCipher::isUpperChar(wchar_t c) {
146 // Проверяем только прописные русские буквы

150inline wchar_t modAlphaCipher::toUpperChar(wchar_t c) {
151 // Преобразуем в прописные русские буквы
152 if (c >= L'а' && c <= L'я') return c - L'а' + L'А';
153 if (c == L'ё') return L'Ё';
154 return c;
155}
